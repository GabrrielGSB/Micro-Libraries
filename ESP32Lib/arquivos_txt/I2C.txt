// I2C.hpp
#pragma once
#ifndef I2C_HPP
#define I2C_HPP

#include "driver/i2c.h"
#include "esp_err.h"
#include "GPIO.hpp"

// Frequências Padrão do Barramento I2C
#define I2C_STANDARD_MODE 100000 // 100 kHz (Para sensores mais antigos/lentos)
#define I2C_FAST_MODE     400000 // 400 kHz (Ideal para o display SSD1306)

class I2C {
protected:
    i2c_port_t porta;

    // Variáveis estáticas globais. 
    // Servem como "memória" para a classe saber se o hardware 
    // já foi configurado por outro dispositivo no mesmo barramento.
    static bool porta_0_inicializada;
    static bool porta_1_inicializada;

public:
    // Construtor: Prepara os pinos e o hardware nativo
    I2C(i2c_port_t porta_i2c, gpio_num_t pino_sda, gpio_num_t pino_scl, uint32_t frequencia_hz = I2C_FAST_MODE);

    // Destrutor: Desliga a porta se o objeto for destruído
    ~I2C();

    // ==========================================
    // MÉTODOS DE COMUNICAÇÃO
    // ==========================================

    // Envia um único byte de comando ou dado
    esp_err_t escrever(uint8_t endereco_dispositivo, uint8_t registo, uint8_t dado);
    
    // Envia um pacote completo de dados (Essencial para desenhar pixels no ecrã SSD1306)
    esp_err_t escrever_bloco(uint8_t endereco_dispositivo, uint8_t registo, const uint8_t *dados, size_t tamanho);

    // Lê informações de sensores (Ex: pedir a temperatura a um sensor I2C)
    esp_err_t ler(uint8_t endereco_dispositivo, uint8_t registo, uint8_t *buffer_recepcao, size_t tamanho);
    
    // Método utilitário para classes externas (como o display) saberem qual porta usar
    i2c_port_t getPorta();
};

#endif


// I2C.cpp
#include "I2C.hpp"
#include "esp_log.h"

// Tag para usarmos nos logs do monitor serial
static const char* TAG = "I2C_HAL";

// Inicialização das variáveis estáticas (essencial no C++)
bool I2C::porta_0_inicializada = false;
bool I2C::porta_1_inicializada = false;

// ==========================================
// CONSTRUTOR E DESTRUTOR
// ==========================================

I2C::I2C(i2c_port_t porta_i2c, gpio_num_t pino_sda, gpio_num_t pino_scl, uint32_t frequencia_hz) {
    this->porta = porta_i2c;

    bool precisa_inicializar = false;

    // Verifica se a porta solicitada já está em uso por outro objeto
    if (porta == I2C_NUM_0 && !porta_0_inicializada) {
        precisa_inicializar = true;
        porta_0_inicializada = true;
    } else if (porta == I2C_NUM_1 && !porta_1_inicializada) {
        precisa_inicializar = true;
        porta_1_inicializada = true;
    }

    // Só reconfigura o hardware físico se for a primeira vez
    if (precisa_inicializar) {
        i2c_config_t conf = {};
        conf.mode = I2C_MODE_MASTER;               // ESP32 será o Mestre do barramento
        conf.sda_io_num = pino_sda;                // Pino de Dados
        conf.scl_io_num = pino_scl;                // Pino de Clock
        conf.sda_pullup_en = GPIO_PULLUP_ENABLE;   // Ativa resistores internos (Requisito I2C)
        conf.scl_pullup_en = GPIO_PULLUP_ENABLE;   
        conf.master.clk_speed = frequencia_hz;     // Velocidade (100kHz ou 400kHz)
        conf.clk_flags = 0;                        // Clock padrão

        // Aplica as configurações aos pinos
        esp_err_t err = i2c_param_config(porta, &conf);
        
        // Se a configuração deu certo, instala o driver no Kernel
        if (err == ESP_OK) {
            err = i2c_driver_install(porta, conf.mode, 0, 0, 0);
            ESP_LOGI(TAG, "Hardware I2C inicializado com sucesso na porta %d", porta);
        }
        else {
            ESP_LOGE(TAG, "Erro fatal ao inicializar I2C na porta %d", porta);
        } 
    } 
    else {
        ESP_LOGI(TAG, "A porta I2C %d já estava aberta. Conectando-se ao barramento existente.", porta);
    }
}

I2C::~I2C() {
    ESP_LOGW(TAG, "Objeto I2C destruído. Driver nativo mantido online para não quebrar outros dispositivos.");
}

// ==========================================
// MÉTODOS DE COMUNICAÇÃO
// ==========================================

esp_err_t I2C::escrever_bloco(uint8_t endereco_dispositivo, uint8_t registrador, const uint8_t *dados, size_t tamanho) {
    // 1. Cria a "caixa" (link) onde vamos empacotar os comandos
    i2c_cmd_handle_t cmd = i2c_cmd_link_create();
    
    // 2. Empacotando os passos do I2C:
    i2c_master_start(cmd); // Sinal de START
    
    // Envia o Endereço do Chip + Bit indicando que queremos ESCREVER (WRITE)
    i2c_master_write_byte(cmd, (endereco_dispositivo << 1) | I2C_MASTER_WRITE, true);
    
    // Envia o Endereço do Registrador interno do display/sensor
    i2c_master_write_byte(cmd, registrador, true);
    
    // Envia a rajada de dados de uma vez (ideal para enviar imagens pro display)
    i2c_master_write(cmd, dados, tamanho, true);
    
    i2c_master_stop(cmd); // Sinal de STOP
    
    // 3. Despacha o pacote completo para o hardware executar (Timeout de 1000ms)
    esp_err_t err = i2c_master_cmd_begin(porta, cmd, 1000 / portTICK_PERIOD_MS);
    
    // 4. Libera a memória da "caixa"
    i2c_cmd_link_delete(cmd);
    
    return err;
}

esp_err_t I2C::escrever(uint8_t endereco_dispositivo, uint8_t registrador, uint8_t dado) {
    // Reaproveitamos a função escrever_bloco, enviando um bloco de tamanho 1
    return escrever_bloco(endereco_dispositivo, registrador, &dado, 1);
}

esp_err_t I2C::ler(uint8_t endereco_dispositivo, uint8_t registrador, uint8_t *buffer_recepcao, size_t tamanho) {
    if (tamanho == 0) return ESP_OK; // Segurança contra quebras

    i2c_cmd_handle_t cmd = i2c_cmd_link_create();
    
    // FASE 1: "Avisar" o chip de qual registrador queremos ler
    i2c_master_start(cmd);
    i2c_master_write_byte(cmd, (endereco_dispositivo << 1) | I2C_MASTER_WRITE, true);
    i2c_master_write_byte(cmd, registrador, true);
    
    // FASE 2: Fazer o "Restart" e mudar para modo LEITURA (READ)
    i2c_master_start(cmd);
    i2c_master_write_byte(cmd, (endereco_dispositivo << 1) | I2C_MASTER_READ, true);
    
    // FASE 3: Ler os bytes
    if (tamanho > 1) {
        // Lê todos os bytes (exceto o último) e responde com ACK (Confirmação)
        i2c_master_read(cmd, buffer_recepcao, tamanho - 1, I2C_MASTER_ACK);
    }
    // Lê o último byte e responde com NACK (Avisando o chip que encerramos a leitura)
    i2c_master_read_byte(cmd, buffer_recepcao + tamanho - 1, I2C_MASTER_NACK);
    
    i2c_master_stop(cmd);
    
    // Executa e limpa
    esp_err_t err = i2c_master_cmd_begin(porta, cmd, 1000 / portTICK_PERIOD_MS);
    i2c_cmd_link_delete(cmd);
    
    return err;
}

i2c_port_t I2C::getPorta() {
    return porta;
}
